\[\text{What is the connection between eigenvalues and similar matrices?}\]	\[\text{Similar matrices have the same eigenvalues:}\]\[B=P^{-1}AP\implies\operatorname{det}(B-\lambda I)=\operatorname{det}(P^{-1}AP-P^{-1}(\lambda I)P)=\operatorname{det}(P^{-1}(A-\lambda I)P)=\boxed{\operatorname{det}(A-\lambda I)=\operatorname{det}(B-\lambda I)}\]
\[\text{Similar matrices always have the same eigenvalues, but do they have the same eigenvectors?}\]	\[\text{No.}\]\[Bv=\lambda v\implies P^{-1}APv=\lambda v\Rightarrow A(Pv)=\lambda(Pv)\implies\boxed{Pv\;\text{is an eigenvector of}\;A}\]
\[\text{Given some transformation and ordered basis}\;\beta,\;\text{how would you determine if}\;\beta\;\text{consists of eigenvectors?}\]	\[\text{Make a matrix}\;[T]_{\beta}\;\text{by transforming all the vectors from}\;\beta\;\text{and rewriting them with respect to}\;\beta\]\[\text{By definition, if}\;\beta\;\text{consists of eigenvectors of}\;T\;\text{then}\;[T]_{\beta}=D\;\text{is a diagonal matrix with eigenvalues in corresponding columns}\]
\[\text{Given some transformation}\;T,\;\text{how would you find an ordered basis}\;\beta\;\text{s.t.}\;[T]_{\beta}\;\text{is diagonal?}\]	\[[T]_{\beta}\;\text{is diagonal}\implies\beta\;\text{consists of eigenvectors of}\;T\]\[\text{We will rewrite transformation as a matrix, find its eigenvalues and solve for eigenvectors}\]\[\text{Example:}\;T(a,b)=(-2a+3b,-10a+9b)\implies A=\begin{pmatrix}-2&3\\-10&9\end{pmatrix}\]
\[\text{What is the relation between transpose and eigenvalues?}\]	\[A\;\text{and}\;A^T\;\text{have the same characteristic polynomials and therefore the same eigenvalues}\]
\[\text{Prove that}\;A\;\text{and}\;A^T\;\text{have the same eigenvalues}\]	\[\text{Same eigenvalues (including multiplicity) means the same characteristic equations}\]\[\operatorname{det}(A-\lambda I)=\operatorname{det}\Big[(A-\lambda I)^T\Big]=\operatorname{det}(A^T-\lambda I)\]
\[\text{If}\;T\;\text{is a linear operator on an}\;n\text{-dimensional vector space}\;V,\;\text{when is it diagonalizable? (related to eigenvalues)}\]	\[T\;\text{is diagonalizable when it has}\;n\;\text{distinct eigenvalues}\]
\[\text{Give an example of a diagonalizable matrix with less than}\;n\;\text{distinct eigenvalues}\]	\[I=\begin{pmatrix}1&0\\0&1\end{pmatrix}\]\[\text{Every vector multiplies by 1}\]
\[\text{What is the Multiplicity?}\]	\[\text{The multiplicity of}\;\lambda\;\text{is the largest positive integer}\;k\;\text{for which}\;(t-\lambda)^k\;\text{is a factor of}\;f(t)\]
\[\text{What is the definition of the Eigenspace?}\]	\[\text{Let}\;T\;\text{be a linear operator on}\;V\;\text{and let}\;\lambda\;\text{be an eigenvalue of}\;T\]\[\text{The set}\;E_{\lambda}=\{x\in V\mid T(x)=\lambda x\}=\operatorname{ker}(T-\lambda I_v)\;\text{is an eigenspace of}\;T\;\text{corresponding to the eigenvalue}\;\lambda\]
\[\text{What is the dimension of the Eigenspace for the eigenvalue}\;\lambda\;\text{of}\;T?\]	\[\text{Let}\;T\;\text{be a linear operator on a finite-dimensional vector space}\;V\;\text{and let}\;\lambda\;\text{be its eigenvalue with multiplicity}\;m\]\[\text{Then:}\quad 1\le\operatorname{dim}(E_{\lambda})\le m\]
