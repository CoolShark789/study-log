\[\text{What is the definition of the bounded sequence?}\]	\[\text{Sequence}\;x_n\;\text{is bounded if}\;\exists M>0\;\text{s.t.}\;\forall n\in\mathbb{N}\,|x_n|\le M\]\[\text{Meaning we can find an interval}\;[-M,M]\;\text{for all the terms of the sequence}\]
\[\text{Prove:}\;\lim(ca_n)=ca\]	\[1)\;c=0\Rightarrow ca=0,\,\forall n\in\mathbb{N}(ca_n)=0.\;\forall\epsilon >0:\;|ca_n-ca|<\epsilon\iff |0-0|<\epsilon\implies \lim(ca_n)=0=ca\]\[2)\;c\ne 0.\;\forall\epsilon >0:\;|ca_n-ca|<\epsilon\Rightarrow |c||a_n -a|<\epsilon\Rightarrow |a_n-a|<\frac{\epsilon}{|c|}\]\[\lim(a_n)=a\Rightarrow\exists N\in\mathbb{N}\;\text{s.t.}\;\forall n\ge N |a_n-a|<\frac{\epsilon}{|c|}\Rightarrow |ca_n-ca|<\epsilon\implies\lim(ca_n)=ca\]\[\hspace{45em}\square\]
\[\text{Prove:}\;\lim(a_n+b_n)=a+b\]	\[\forall\epsilon >0:\;|(a_n+b_n)-(a+b)|<\epsilon,\;|(a_n-a)+(b_n-b)|\le |a_n-a|+|b_n-b|<\epsilon\]\[\text{Since}\;a_n\rightarrow a,\;b_n\rightarrow b:\quad1)\;\exists N_1\in\mathbb{N}\;\text{s.t.}\;\forall n\ge N_1\;|a_n -a|<\frac{\epsilon}{2},\quad\text{2)}\;\exists N_2\in\mathbb{N}\;\text{s.t.}\;\forall n\ge N_2\;|b_n -b|<\frac{\epsilon}{2}\]\[\text{If}\;N=\max\{N_1, N_2\},\;\text{then}\;\forall n\ge N:\quad|(a_n-a)+(b_n-b)|\le |a_n-a|+|b_n-b|<\frac{\epsilon}{2}+\frac{\epsilon}{2}=\epsilon\]\[\hspace{50em}\square\]
\[\text{Prove:}\;\lim(a_nb_n)=ab\]	\[\forall\epsilon >0:\quad |a_nb_n-ab|=|a_nb_n-b_na+b_na-ab|=|b_n(a_n-a)+a(b_n-b)|\le |b_n||a_n-a|+|a||b_n-b|<\epsilon\]\[\text{Since}\;b_n\rightarrow b\;\text{is convergent it is also bounded:}\;|b_n|\le M.\quad\forall\epsilon >0\exists N_1\in\mathbb{N}\;\text{s.t.}\;\forall n\ge N_1\;|a_n-a|<\frac{\epsilon}{2M}\]\[\exists N_2\in\mathbb{N}\;\text{s.t.}\;\forall n\ge N_2 |b_n-b|<\frac{\epsilon}{2|a|}.\;N=\max\{N_1,N_2\},\;\text{then}\;\forall n\ge N:\]\[|a_nb_n-ab|\le|b_n||a_n-a|+|a||b_n-b|< M \cdot\frac{\epsilon}{2M}+|a|\frac{\epsilon}{2|a|}=\epsilon\]\[\hspace{50em}\square\]
\[\text{What does Algebraic Limit Theorem state?}\]	\[\text{Let}\;\lim a_n=a\;\text{and}\;\lim b_n=b,\;\text{then:}\]\[1.\;\lim(ca_n)=ca\;(\forall c\in\mathbb{R})\quad\quad\quad 2.\;\lim(a_n+b_n)=a+b\]\[3.\;\lim(a_nb_n)=ab\quad\quad\quad 4.\;\lim\left(\frac{a_n}{b_n}\right)=\frac{a}{b}\;(b\ne 0)\]
\[\text{What does Order Limit Theorem state?}\]	\[\text{Assume}\;\lim a_n=a\;\text{and}\;\lim b_n=b\]\[1.\;\forall n\in\mathbb{N}:\;a_n\ge 0\rightarrow a\ge 0\quad\quad\quad 2.\;\forall n\in\mathbb{N}:\;a_n\le b_n\rightarrow a\le b\]\[3.\;\exists c\in\mathbb{R}\forall n\in\mathbb{N}:\;c\le b_n\rightarrow c\le b\quad\text{or}\quad a_n\le c\rightarrow a\le c\]
