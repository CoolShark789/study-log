\[\text{What is the definition of a velocity relative to a moving frame?}\]	\[v^{\operatorname{COM}}_i=v_i-V_{\operatorname{COM}}\]\[v_i\;\text{is the velocity in the lab frame}\quad\quad\quad V_{\operatorname{COM}}\;\text{is the velocity of the center of mass}\]
\[\text{Prove that the momentum in the center of mass frame (COM) is always zero}\]	\[V_{\operatorname{COM}}=\frac{\sum_{i=1}^n m_iv_i}{\sum_{i=1}^n m_i}=\frac{P_{\operatorname{total}}}{M}\quad\quad\quad v^{\operatorname{COM}}_i=v_i-V_{\operatorname{COM}}\implies\sum_{i=1}^n m_iv^{\operatorname{COM}}_i=\sum_{i=1}^n m_i(v_i-V_{\operatorname{COM}})\]\[\sum_{i=1}^n m_iv^{\operatorname{COM}}_i=\sum_{i=1}^n m_iv_i-\sum_{i=1}^n m_iV_{\operatorname{COM}}=P_{\operatorname{total}}-MV_{\operatorname{COM}}=P_{\operatorname{total}}-\cancel{M}\frac{P_{\operatorname{total}}}{\cancel{M}}=0\]\[\boxed{\sum_{i=1}^n m_iv^{\operatorname{COM}}_i=0}\]
\[\text{Show that in elastic collisions for COM coordinates the speed after collision is the same}\]	\[\text{COM:}\quad m_1v_{1,\operatorname{COM}}+m_2v_{2,\operatorname{COM}}=m_1v^{'}_{1,\operatorname{COM}}+m_2v^{'}_{2,\operatorname{COM}}=0\implies v_{2,\operatorname{COM}}=-\frac{m_1}{m_2}v_{1,\operatorname{COM}}\]\[\text{Elastic:}\quad\cancel{\frac{1}{2}}m_1v^{2}_{1,\operatorname{COM}}+\cancel{\frac{1}{2}}m_2v^{2}_{2,\operatorname{COM}}=\cancel{\frac{1}{2}}m_1v^{'2}_{1,\operatorname{COM}}+\cancel{\frac{1}{2}}m_2v^{'2}_{2,\operatorname{COM}}\]\[m_1v^{2}_{1,\operatorname{COM}}+\cancel{m_2}\frac{m^2_1}{m^{\cancel{2}}_2}v^{2}_{1,\operatorname{COM}}=m_1v^{'2}_{1,\operatorname{COM}}+\cancel{m_2}\frac{m^2_1}{m^{\cancel{2}}_2}v^{'2}_{1,\operatorname{COM}}\implies\]\[\implies \left(m_1 +\frac{m^2_1}{m_2}\right)v^{2}_{1,\operatorname{COM}}=\left(m_1 +\frac{m^2_1}{m_2}\right)v^{'2}_{1,\operatorname{COM}}\implies\boxed{\Big|v_{1,\operatorname{COM}}\Big|=\Big|v^{'}_{1,\operatorname{COM}}\Big|,\;\Big|v_{2,\operatorname{COM}}\Big|=\Big|v^{'}_{2,\operatorname{COM}}\Big|}\]
