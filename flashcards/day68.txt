\[\text{Let}\;V,W\;\text{be finite-dimensional vector spaces. Prove:}\;V\;\text{is isomorphic to}\;W\iff\operatorname{dim}(V)=\operatorname{dim}(W)\]	\[(\rightarrow)\;\text{If}\;V\;\text{is isomorhic to}\;W\;\text{then}\;T:V\rightarrow W\;\text{is an isomorphism and is invertible by definition}\]\[\text{For an invertible}\;T:\quad\boxed{\operatorname{dim}(V)=\operatorname{dim}(W)}\]\[(\leftarrow)\;\text{Suppose}\;\operatorname{dim}(V)=\operatorname{dim}(W).\;\text{Let}\;\beta=\{v_1,\cdots ,v_n\}\;\text{and}\;\gamma =\{w_1,\cdots ,w_n\}\;\text{be the bases of}\;V\;\text{and}\;W\]\[\text{There exists a unique linear transformation s.t.}\;T(v_i)=w_i\implies R(T)=\operatorname{span}(T(\beta))=\operatorname{span}(\gamma)=W\]\[\text{Therefore we can conclude that}\;T\;\text{is onto. Because}\;\operatorname{dim}(V)=\operatorname{dim}(W),\;\text{it is also one-to-one}\implies\boxed{T\;\text{is isomorphism}}\]
\[\text{What is the function with domain of a linear transformation and matrix as a range?}\]	\[\text{Let}\;V,W\;\text{be finite-dimensional vector spaces over}\;F\;\text{s.t.}\;\operatorname{dim}(V)=n,\;\operatorname{dim}(W)=m\;\text{and}\;\beta,\gamma\;\text{are their ordered bases}\]\[\text{Then}\;\Phi:\mathcal{L}(V,W)\rightarrow M_{m\times n}(F)\;\text{that is defined as}\;\Phi (T)=[T]_{\beta}^{\gamma}\;\text{for}\;T\in\mathcal{L}(V,W)\;\text{is an isomorphism}\]
\[\text{What is the function for standard representation of}\;V\;\text{with respect to}\;\beta?\]	\[\phi_{\beta}:V\rightarrow\mathbb{F}^n\;\text{defined as}\;\phi_{\beta}(x)=[x]_{\beta}\;\forall x\in V\]
\[\text{What is the change of coordinates matrix? (for example, for bases}\;\beta,\gamma)\]	\[\text{For}\;\beta=\{v_1,\cdots ,v_n\}:\;\text{matrix in the form of}\quad Q=\Big[[v_1]_{\gamma}\cdots [v_n]_{\gamma}\Big]\]\[[v]_{\gamma}=Q[v]_{\beta}\]
\[\text{What is a Linear Operator?}\]	\[\text{Linear Operator is a linear transformation that maps a vector space}\;V\;\text{into itself}\]
\[\text{What is the relation between a linear operator and change of coordinates matrix?}\]	\[\text{Let}\;T\;\text{be a linear operator on a finite-dimensional vector space}\;V.\]\[\text{Suppose}\;Q\;\text{is the change-of-coordinate matrix from}\;\beta\;\text{into}\;\gamma\;\text{then:}\]\[[T]_{\gamma}=Q^{-1}[T]_{\beta}Q\]\[\text{Reminder:}\;[T]_{\gamma}\;\text{is the matrix}\;\Big[[T(w_1)]_{\gamma},\cdots ,[T(w_n)]_{\gamma} \Big]\;\text{where}\;w_n\in\gamma\;\text{and}\;Q\;\text{is}\;\Big[[v_1]_{\gamma},\cdots ,[v_n]_{\gamma}\Big]\;\text{where}\;v_n\in\beta\]
\[\text{Find the transformation}\;T(a,b)\;\text{for any}\;(a,b)\in\mathbb{R}^2\;\text{that will be reflected over}\;y=2x\]	\[\text{Let}\;\beta,\beta '\;\text{be ordered bases:}\;\beta=\{(1,0),(0,1)\},\;\beta '=\{(1,2),(-2,1)\}\]\[(1,2)\;\text{lays on}\;y\;\text{and}\;(-2,1)\;\text{is perpendicular therefore:}\;T(1,2)=(1,2),\;T(-2,1)=(2,-1)\]\[[T]_{\beta '}=\Big[[T(1,2)]_{\beta '}\;[T(-2, 1)]_{\beta '}\Big]=\begin{pmatrix}1&0\\0&-1\end{pmatrix}.\;\text{Change-of-coordinate matrix:}\;Q=\Big[(1,2)_{\beta}\;(-2,1)_{\beta}\Big]=\begin{pmatrix}1&2\\-2&1\end{pmatrix}\]\[[T]_{\beta}=Q[T]_{\beta '}Q^{-1}=\frac{1}{5}\begin{pmatrix}1&2\\-2&1\end{pmatrix}\begin{pmatrix}1&0\\0&-1\end{pmatrix}\begin{pmatrix}1&-2\\2&1\end{pmatrix}=\frac{1}{5}\begin{pmatrix}-3&4\\4&3\end{pmatrix}\implies\boxed{T\begin{pmatrix}a\\b\end{pmatrix}=\frac{1}{5}\begin{pmatrix}-3&4\\4&3\end{pmatrix}\begin{pmatrix}a\\b\end{pmatrix}=\frac{1}{5}\begin{pmatrix}-3a+4b\\4a+3b\end{pmatrix}}\]
