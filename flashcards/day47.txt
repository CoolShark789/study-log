\[\text{What is an Inertial Frame of Reference?}\]	\[\text{Coordinate systems that have no acceleration and for which Newton's first law holds}\]\[\text{Note: We Consider Earth to be an Inertial Frame, because the effects of its rotation\movement are negligible}\]
\[\text{Prove that any system moving uniformly relative to inertial system is also inertial}\]	\[{Let}\;S\;\text{with position vector}\;\mathbf{r}(x,y,z)\;\text{to be inertial and let}\;S'\;\text{to be other coordinate system with a position vector}\;\mathbf{r'}(x',y',z')\]\[\text{Then for some arbitrary point in space:}\;\mathbf{r}=\mathbf{R}+\mathbf{r'}\;\text{where}\;\mathbf{R}\;\text{is the vector connecting the origins of two coordinate systems}\]\[\text{Differentiate both sides two times and multiply by}\;m:\quad m\mathbf{\ddot r}=\mathbf{F}_{true}=m\mathbf{\ddot R}+m\mathbf{\ddot r'}\implies\mathbf{F}_{apparent}=\mathbf{F}_{true}-m\mathbf{\ddot R}\]\[\text{Since}\;S'\;\text{moves uniformly:}\;\mathbf{\ddot R}=0\Rightarrow \mathbf{F}_{apparent}=\mathbf{F}_{true}\implies \boxed{S'\;\text{is inertial since Newton's Laws hold there}}\]
\[\text{What is a Fictitious Force?}\]	\[\text{A force that appears in the description of a motion in a non-intertial system}\]\[\text{For this force there is no interaction involved and is only felt when non inertial frame of reference is involved, thus the name}\]
\[\text{What does}\;\mathbf{F}_{\text{fictitious}}\;\text{equal to? (2 equations)}\]	\[1.\;\mathbf{F}_{\text{fictitious}}=-M\mathbf{\ddot R}\quad\;(\mathbf{\ddot R}\;\text{is the acceleration of non inertial system with respect to the inertial one)}\]\[2.\;\mathbf{F}_{\text{apparent}}=\mathbf{F}_{\text{real}}+\mathbf{F}_{\text{fictitious}}\]
\[\text{How to calculate acceleration in Polar Coordinates?}\]	\[\mathbf{a}=\frac{d\mathbf{v}}{dt}=\frac{d}{dt}(\dot{\mathbf{r}}\hat{\mathbf{r}}+r\dot{\mathbf{\theta}}\hat{\mathbf{\theta}})=\ddot{\mathbf{r}}\hat{\mathbf{r}}+\dot{\mathbf{r}}\dot{\mathbf{\theta}}\hat{\mathbf{\theta}}+r\ddot{\mathbf{\theta}}\hat{\mathbf{\theta}}-r\dot{\mathbf{\theta}}^2\hat{\mathbf{r}}=(\ddot{\mathbf{r}}-r\dot{\mathbf{\theta}}^2)\hat{\mathbf{r}}+(\dot{\mathbf{r}}\dot{\mathbf{\theta}}+r\ddot{\mathbf{\theta}})\hat{\mathbf{\theta}}=\mathbf{a}_r\hat{\mathbf{r}}+a_{\theta}\hat{\mathbf{\theta}}\]
