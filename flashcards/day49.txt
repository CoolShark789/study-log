\[\text{What is Momentum equal to and its relation to the Newton's Second Law?}\]	\[\mathbf{F}=M\mathbf{a}=M\frac{d\mathbf{v}}{dt}=\frac{d}{dt}(M\mathbf{v})=\frac{d\mathbf{P}}{dt}\quad(\mathbf{P}=M\mathbf{v})\]
\[\text{Derive the relation between total momentum and the total external force}\]	\[\text{If there is some system of particles then:}\quad f_j ^{\text{int}}+f_j ^{\text{ext}}=f_j=\frac{d\mathbf{p}_j}{dt}\]\[\text{For all the particles:}\quad\sum_{j=1}^N f_j ^{\text{int}}+\sum_{j=1}^N f_j ^{\text{ext}}=\sum_{j=1}^N\frac{d\mathbf{p}_j}{dt}=\frac{d}{dt}\sum_{j=1}^N\mathbf{p}_j=\frac{d\mathbf{P}_{\text{total}}}{dt}\]\[\text{By the Newton's Third Law for each pair of particles there is a pair of equal and opposite forces:}\;\sum_{j=1}^N f_j ^{\text{int}}=0\]\[\sum_{j=1}^N f_j ^{\text{ext}}=\boxed{F_{\text{total}}=\frac{d\mathbf{P}_{\text{total}}}{dt}}\]
\[\text{How to find the center of mass? (simple, without integration)}\]	\[\mathbf{F}=M\ddot{\mathbf{R}}=\frac{d\mathbf{P}}{dt}=\sum_{j=1}^N \frac{d}{dt}(m_j\dot{\mathbf{r}}_j)=\sum_{j=1}^N m_j\ddot{\mathbf{r}}_j\]\[\text{Then}\;M\mathbf{R}=\sum_{j=1}^N m_j\mathbf{r}_j\implies\mathbf{R}=\frac{1}{M}\sum_{j=1}^N m_j\mathbf{r}_j\]\[\text{Where}\;M\;\text{is the total mass and}\;\mathbf{R}\;\text{is the vector from origin to the center of mass}\]
\[\text{What is the general formula for the Center of the Mass?}\]	\[\mathbf{R}=\frac{1}{M}\sum_{j=1}^N m_j\mathbf{r}_j\implies\lim_{n\to\infty}\mathbf{R}=\frac{1}{M}\int_{\text{V}}\mathbf{r}\,dm=\frac{1}{M}\int_{\text{V}}\mathbf{r}\rho\,dV\]
