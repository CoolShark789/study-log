\[\text{Find the speed of the bullet by analyzing its collision with the block on a string}\]	\[mV=(m+M)V'\Rightarrow V'=\frac{m}{m+M}V\quad E_i=\frac{1}{2}(m+M)V'^2,\;E_f=(m+M)gh\]\[\cos\phi=\frac{l-h}{l}\Rightarrow h=l(1-\cos\phi)\quad E_i=E_f:\;\frac{1}{2}\cancel{(m+M)}\frac{m^2}{(m+M)^{\cancel{2}}}V^2=(m+M)gl(1-\cos\phi)\]\[V=\sqrt{\frac{2gl(m+M)^2(1-\cos\phi)}{m^2}}=\left(\frac{m+M}{m}\right)\sqrt{2gl(1-\cos\phi)}\]
\[\text{Mass}\;m\;\text{slides from a top of a sphere on a frictionless surface. At what angle and distance from the top will it fall?}\]	\[\frac{mv^2}{R}=mg\cos\theta-N\implies\text{Mass falls at}\;N=0:\;\frac{v^2}{R}=g\cos\theta\Rightarrow v^2=gR\cos\theta\]\[E_i=mgR,\;E_f=\frac{1}{2}mv^2+mgR\cos\theta\quad E_i=E_f:\;2\cdot \cancel{m}gR=\cancel{2\cdot\frac{1}{2}m}v^2+2\cdot\cancel{m}gR\cos\theta\]\[2gR=v^2+2gR\cos\theta=3gR\cos\theta\implies\cos\theta=\frac{2}{3},\;\boxed{\theta\approx 48.19^{\degree}}\implies R-R\cos\theta=\boxed{\frac{1}{3}R}\]
\[\text{Why any oscillating motion can be thought of as a SHM?}\]	\[\text{Taylor Series expansion for}\;U(r)\;\text{around}\;r_0:\quad U(r)=U(r_0)+(r-r_0)\frac{dU}{dr}\Big|_{r_0}+\frac{1}{2}(r-r_0)^2\frac{d^2U}{dr^2}\Big|_{r_0}+\cdots\]\[U(r)\approx U(r_0)+(r-r_0)\frac{dU}{dr}\Big|_{r_0}+\frac{1}{2}(r-r_0)^2\frac{d^2U}{dr^2}\Big|_{r_0}.\;\text{Potential energy is at its minimum at equilibrium:}\;\frac{dU}{dr}\Big|_{r_0}=0\]\[\boxed{U(r)\approx\operatorname{const}+\frac{1}{2}(r-r_0)^2\frac{d^2U}{dr^2}\Big|_{r_0}\quad U(x)=\operatorname{const}+\frac{1}{2}k(x-x_0)^2}\implies\frac{d^2U}{dr^2}\Big|_{r_0}=k\]
\[\text{What does}\;k\;\text{equal to in SHM?}\]	\[k=\frac{d^2U}{dr^2}\Big|_{r_0}\]
\[\text{What is the Angular Frequency of a SHM where potential energy has a quadratic form?}\]	\[U=\frac{1}{2}Aq^2+\operatorname{const},\quad K=\frac{1}{2}B\dot{q}^2\implies E=U+K=\operatorname{const}\;\text{(system is conservative)}\]\[\frac{dE}{dt}=A\dot{q}q+B\dot{q}\ddot{q}=0\Rightarrow\ddot{q}+\frac{A\cancel{\dot{q}}}{B\cancel{\dot{q}}}q=0\implies\boxed{\omega=\sqrt{\frac{A}{B}}}\]
\[\text{Find angular frequency of a pendulum}\]	\[U=mgl(1-\cos\theta)\approx\frac{1}{2}mgl\theta^2,\quad K=\frac{1}{2}ml^2\dot{\theta}^2\]\[\text{Quadratic Energy Form:}\quad\omega=\sqrt{\frac{\cancel{m}g\cancel{l}}{\cancel{m}l^{\cancel{2}}}}=\sqrt{\frac{g}{l}}\]
\[\text{When is equilibrium stable, unstable, neutral and why?}\]	\[\text{1. Stable}:\;\frac{d^2U}{dx^2}>0\quad\quad\quad\text{2. Unstable}:\;\frac{d^2U}{dx^2}<0\quad\quad\quad\text{3. Neutral}:\;\frac{d^2U}{dx^2}=0\]\[\frac{d^2U}{dx^2}>0\;\text{means that}\;\frac{dU}{dx}\;\text{is at minimum and positive displacement will cause negative force, returning position to equilibrium}\]\[\text{On the other hand (for}\;\frac{d^2U}{dx^2}<0)\;\text{positive displacement will cause positive force pushing position out of equilibrium}\]\[\text{Reminder:}\;F=-\frac{dU}{dx}\]
