\[\text{Define Pointwise Convergence}\]	\[\forall n\in\mathbb{N}\;\text{let's define a function}\;f_n:A\rightarrow\mathbb{R}\]\[\text{The sequence}\;f_n\;\text{of functions converges pointwise on}\;A\;\text{to a function}\;f\;\text{if}\;\forall x\in A\;\lim_{n\to \infty}f_n(x)=f(x)\]\[\text{It can be rewritten as:}\;\forall x\in A\;\forall\epsilon >0\;\exists N=N(x,\epsilon)\in\mathbb{N}\;\text{s.t.}\;\forall n\ge N:\;|f_n(x)-f(x)|<\epsilon\]
\[\text{Why is Pointwise Convergence called that?}\]	\[\text{Because we check point by point. Recall the logical definition:}\]\[\forall x\in A\;\forall\epsilon >0\;\exists N=N(x,\epsilon)\in\mathbb{N}\;\text{s.t.}\;\forall n\ge N:\;|f_n(x)-f(x)|<\epsilon\]\[\text{Here we are picking each point from the domain for each}\;f_n\;\text{and choose such}\;N\;\text{that}\;f_n(x)\;\text{is in the}\;\epsilon\text{-neighborhood}\]\[\text{Unlike Uniform Convergence - we will need different}\;N\;\text{for different points, so it is dependent on}\;x\;\text{and is checked pointwise}\]
\[\text{What is the definition of the Uniform Convergence?}\]	\[\text{Let}\;f_n\;\text{be a sequence of functions defined as}\;f_n:A\rightarrow\mathbb{R}\]\[f_n\;\text{is uniformly convergent on}\;A\;\text{if}\;\forall\epsilon>0\;\exists N=N(\epsilon)\in\mathbb{N}\;\text{s.t.}\;\forall x\in A\;\forall n\ge N:\;|f_n(x)-f(x)|<\epsilon\]
\[\text{What's the main difference between Pointwise and Uniform Convergence?}\]	\[\text{Pointwise convergence has a response}\;N,\;\text{after which the condition is true, dependent on}\;x\;\text{and}\;\epsilon\]\[\text{While Uniform Convergence has the same}\;N\;\text{for all}\;x\]
\[\text{Show that}\;g_n(x)=\frac{1}{n(1+x^2)}\;\text{is uniformly convergent to}\;g(x)\]	\[g(x)=\lim_{n\to\infty}g_n(x)=0\implies|g_n(x)-g(x)|=\left|\frac{1}{n(1+x^2)}-0\right|\le\frac{1}{n}<\epsilon\]\[\text{Let}\;N>\frac{1}{\epsilon}\;\text{then}\;\forall\epsilon >0\;\forall x\in A\;\forall n\ge N:\; n>\frac{1}{\epsilon}\Rightarrow\epsilon >\frac{1}{n}\ge\frac{1}{n(1+x)^2}\implies\boxed{\left|\frac{1}{n(1+x^2)}-0\right|=|g_n(x)-g(x)|<\epsilon}\]
\[\text{Show that}\;f_n(x)=\frac{x^2+nx}{n}\;\text{isn't uniformly convergent to}\;f(x)\]	\[f(x)=\lim_{n\to\infty}f_n(x)=x\implies|f_n(x)-f(x)|=\left|\frac{x^2+\cancel{nx}}{n}-\cancel{x}\right|=\frac{x^2}{n}<\epsilon\]\[\text{Then we have to choose}\;N>\frac{x^2}{\epsilon}\;\text{meaning}\;N\;\text{is different for different}\;x\]
\[\text{What does Cauchy Criterion for Uniform Convergence state?}\]	\[\text{A sequence of functions}\;f_n:A\rightarrow\mathbb{R}\;\text{is uniformly convergent on}\;A\;\text{to}\;f\iff\]\[\iff\forall\epsilon>0\;\exists N\in\mathbb{N}\;\text{s.t.}\;\forall x\in A\;\forall m,n\ge N:\;|f_n(x)-f_m(x)|<\epsilon\]
